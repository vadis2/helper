    \section{Properties} \label{sec:Properties}

    \begin{equation}
        C^k_n = C^{n-k}_n
    \end{equation}

    \begin{equation}
        C^k_{n} = C^k_{n-1} + C^{k-1}_{n-1}
    \end{equation}

    \begin{equation}
        \sum_{k=0}^n C^k_n = 2^n
    \end{equation}

    \begin{equation}
        \sum_{k=0}^n (C^k_{n})^2 = C^k_{2n}
    \end{equation}

    \begin{equation}
        C^n_{n+m} = \sum_{k=0}^m C^{n-1}_{n+k-1}
    \end{equation}

    \begin{equation}
        \sum_{k=0}^m m^2 = \frac{m(m+1)(2m+1)}{6}
    \end{equation}

    \begin{equation}
        C^0_n - C^1_n + C^2_n + ... + (-1)^n C^n_n = \left\{
        \begin{array}{rl}
            1 & \text{if } n = 0,\\
            0 & \text{if } n \geq 0.
        \end{array} \right.
    \end{equation}
